#include<stdio.h>
#include<math.h>
#include<stdlib.h>
int gcd(int a, int b)
{
    int i,va,vb;
    va=abs(a),vb=abs(b);
    int min=va<vb?va:vb;
    if(a==0) return vb;
    else if(b==0) return va;
    else
    {
        for(i=min;i>1;i--)
        {
            if(va%i==0&&vb%i==0) break;
        }
        return i;
    }
    
}
int main()
{
    int a,b;
    scanf("%d %d",&a,&b);
    printf("%d\n",gcd(a,b));
    return 0;
}

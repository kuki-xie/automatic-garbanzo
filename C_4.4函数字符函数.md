#include<stdio.h>
int main(void)
{
    char a[500];
    int i,j,sum=0;
    for(i = 0;i < 500;i++)
    {
        scanf("%c",&a[i]);
        if(a[i] == '\n') break;
        else if(a[i] >= 'a' && a[i] <= 'z') a[i] -= 32;
        else if(a[i] >= 'A' && a[i] <= 'Z') a[i] += 32;
        else if(a[i] >= '0' && a[i] <= '9') sum++;
    }
    printf("%d\n",sum);
    for(j = 0;j <= i;j++)
    printf("%c",a[j]);
    return 0;
}

#include<stdio.h>
#include<math.h>
int fun(int a,int b)
{
	int i,n,k=0,t,sum;
	for(i=a;i<=b;i++)
	{
		t=i;
		n=0;
		while(t)
		{
			t/=10;
			n++;
		}
		t=i;
		if(n>=3)
		{
			sum=0;
			while(t)
			{
				sum+=pow(t%10,n);
				t/=10;
			}
			if(sum==i)
			{
				k++;
			}
		}
	}
	return k;
}
int main()
{
	int m,i,a,b,c[100];
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		scanf("%d %d",&a,&b);
		c[i]=fun(a,b);
	}
	for(i=0;i<m;i++)
	{
		printf("%d ",c[i]);
	}
	return 0;
}

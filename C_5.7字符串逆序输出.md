void inv(char *x,int n){
	char *i,*j,*p;int m;
	m=(n-1)/2;
	i=x;j=x+n-1;p=x+m;
	for(i=x;i<=p;i++,j--){
		char temp=*i;
		*i=*j;
		*j=temp;
	}
}

#include<stdio.h>
#include<string.h>
int main()
{
    int i,j,flag;char str[51],ch;
    int n1,n2;
    scanf("%s",str);
    int n=strlen(str);
    for(i=0;i<n;i++)
    {
        flag=0;
        if((str[i]>='a'&&str[i]<='z')||(str[i]>='A'&&str[i]<='Z'))
        {
            printf("%c",str[i]);
            ch=str[i];
        }
        else
        {
            n1=str[i]-'0';
            if(i<=n-2 && str[i+1]>='0'&&str[i+1]<='9')
            {
                flag=1;
                n2=str[i+1]-'0';
            }
            if(flag==1)  {n1=n1*10+n2;i++;}
            for(j=0;j<n1-1;j++)
                printf("%c",ch);
        }
    }
    printf("\n");
    return 0;
}

输入说明

输出说明
在一行上输出解压缩后的英文字符串（长度不超过100），最后换行。
输入样例
样例1输入
a5b3aba13b4
输出样例
样例1输出
aaaaabbbabaaaaaaaaaaaaabbbb  */

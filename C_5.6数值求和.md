#include<stdio.h>
int main(){
    int i,j,n,sum=0,a[20][20];
    scanf("%d",&n);
    for(i=0;i<n;i++){
        for(j=0;j<n;j++){
            scanf("%d",&a[i][j]);
        }
    }
    for(i=0;i<n;i++){
        sum+=(a[i][i]+a[i][n-i-1]);
    }
    if(n%2==1)  sum=sum-a[n/2][n/2];
    printf("%d",sum);
    return 0;
}  
/* 问题描述
输入整数N（N≤20），输入二维矩阵A[N] [N]各元素值，计算所有对角线元素的和，每一元素只加一次，输出结算结果。
输入说明
输入整形数据N和N×N矩阵各元素值，如2和矩阵元素A={1,2,3,4}。
输出说明
格式输出：输出计算结果。
输入样例
2
1 2
3 4
输出样例
10 */

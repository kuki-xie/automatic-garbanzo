int main()
{
	int i,num=1;
	char a[109];
	for(i=0;i<100;i++){
		scanf("%c",&a[i]);
		if(a[i]=='\n') break;
	}
	int n=i;
	for(i=0;i<n;i++)
		if(a[i]!=' '&&a[i+1]==' '){
			num++;
		}
	printf("%d",num);
	return 0;
}

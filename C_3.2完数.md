```c
#include<stdio.h>
int isPERFECTNUMBER(int n)
{
    int i,sum=0;

    for(i=1;i<=n/2;i++)
    {
        if(n%i==0)
        sum+=i;
    }
    return sum==n;
}
int main()
{
    int a,b,n;
    scanf("%d %d",&a,&b);
    for(n=a;n<=b;n++)
    {
        if(isPERFECTNUMBER(n))
        {
            printf("%d\n",n);
        }
    }
    return 0;
}
```

{int i,j,a[100];
 for(i = 0;i >= 0;i++)
 {
    scanf("%d",&a[i]);

 }
 int n=i,temp;
 for(i = 0;i < n;i++)
 {
    for(j = 0;j < n-i;j++)
    {
        if(a[j] > a[j+1])
        {
            temp=a[j];
            a[j]=a[j+1];        //先排序
            a[j+1]=temp;
        }
    }
 }


 int t=a[0],count;
 for(i=0;i<=n+1;)
 {
    count=0;
    for(j=i;j<=n+1;j++)   //请教大哥得

        if(t==a[j])
        count++;
        else
        {
            printf("%d %d\n",t,count);
            break;
        }
    }
    i=j;
    t=a[j];     
 }

 return 0;
}

#include<stdio.h>
int gcd(int a,int b)
{
    int i;
    int min=a<b?a:b;

    if (a==0) return b;
    else if (b==0) return a;
    else
    {
        for (i = min; i > 1; i--)
        {
            if (a%i==0&&b%i==0)
            break;
        }
        return i;
    }
}
int main(int argc, char const *argv[])
{
    int a,b;
    while(1)
    {
        scanf("%d %d",&a,&b);
        if(a==0&&b==0)
        break;
        printf("%d\n",gcd(a,b));
    }
    return 0;
}

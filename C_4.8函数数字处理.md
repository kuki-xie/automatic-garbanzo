#include<stdio.h>
long num(int x)
{
    int i,sum = 0;
    for(i = 0;i >= 0;i++)
    {
        sum = sum + x % 10;
        x /= 10;
        if(x == 0) break;
    }
    if(sum > 9) return num(sum);
    else return(sum);
}

int main(void)
{
    int i,n;
    long a[1000000];
    for(i = 0; i >= 0;i++)
    {
        scanf("%ld",&a[i]);
        if(a[i] == 0) break;
    }
    n=i;
    for(i = 0;i < n;i++)
    {
        printf("%ld\n",num(a[i]));
    }
    return 0;
   
    
}

输入说明
多组数据，每行一个整数，输入为0时表示输入结束。
输出说明
输出每个整数的处理结果，然后换行。
输入样例

#include<stdio.h>
#include<string.h>
void sort(char a[],int n){
	int i,j;
	for(j=0;j<n-1;j++){
		for(i=0;i<n-j-1;i++){
			if(a[i]>a[i+1]){
			char t=a[i];
			a[i]=a[i+1];
			a[i+1]=t;
		 	}
		}
	}
}
int main(){
	char a[80];
	int n;
	while(1){
		scanf("%s",a);
		n=strlen(a);
		if(n==0)	break;
		sort(a,n);
		printf("%s\n",a);
	}
	return 0;
}

#include<stdio.h>
#include<math.h>
int ABS(int x)
{
    int y;
    if(x==0) y=0;
    else if(x>0) y=x;
    else y=-x;
    return y;
}

int main()
{
    int n,i,j,max;
    int a[31];
    scanf("%d\n",&n);

    for ( i = 0; i < n; i++)
    {
        scanf("%d",&a[i]);
    }
    max=ABS(a[1]-a[0]);
    for ( i = 0; i < n-1; i++)
    {
        int temp=ABS(a[i+1]-a[i]);
        if(max<temp) max=temp;
    }    
    printf("%d\n",max);
    return 0;    
}
